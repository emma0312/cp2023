2.

#include <stdio.h>
float circle( int r ); /* 宣告 circle 的 prototype */
void main()
{
float answer;
answer = circle(8);
printf( " 圓周長度是 %f", answer );
}
/* ====================
 circle 函數, 計算 circle 的圓周長
==================== */
float circle( int r )
{
float result;
result = 3.14159 * (double)2 * r;
return ( result );
}

5.

#include <stdio.h>
void main()
{
int i, j;
j = 10; /* 迴圈外先設定初值 */
for ( i = 0; i < 6; i++ )
{
printf( "i = %d, ", i );
printf( "j = %d \n", j );
j++; /* 迴圈後加上變更運算式 */
}
}
